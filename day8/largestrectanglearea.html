// LeetCode84 JS单调栈标准解法 function largestRectangleArea(heights) { const stack = []; let maxArea = 0; // 末尾补哨兵0 const arr = [...heights, 0]; for(let i = 0; i < arr.length; i++){ // 当前高度小于栈顶高度，弹出计算 while(stack.length && arr[i] < arr[stack.at(-1)]){ const curIdx = stack.pop(); const h = arr[curIdx]; const left = stack.length ? stack.at(-1) : -1; const w = i - left - 1; maxArea = Math.max(maxArea, h * w); } stack.push(i); } return maxArea; } // 样例调用 largestRectangleArea([2,1,5,6,2,3])
