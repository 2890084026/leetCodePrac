一、删除链表中倒数第 N 个节点                                                                       
输入：head = [1,2,3,4,5], n = 2
输出：[1,2,3,5]
示例 2：
输入：head = [1], n = 1
输出：[]
示例 3：
输入：head = [1,2], n = 1
输出：[1]
 
解法：快慢双指针，快指针先走 N 步，然后两个指针同时走，当快指针到末尾时，慢指针刚好在倒数第 N+1 个节点。
function removeNthFromEnd(head, n) {
    const dummy = { val: 0, next: head };
    let fast = dummy, slow = dummy;

    // 快指针先走 n+1 步
    for (let i = 0; i <= n; i++) {
        fast = fast.next;
    }

    // 同步走直到快指针到末尾
    while (fast !== null) {
        fast = fast.next;
        slow = slow.next;
    }

    // 删除目标节点
    slow.next = slow.next.next;
    return dummy.next;
}

二. 识别有效括号
给定一个只包括 '('，')'，'{'，'}'，'['，']' 的字符串 s ，判断字符串是否有效。
有效字符串需满足：
1. 左括号必须用相同类型的右括号闭合。
2. 左括号必须以正确的顺序闭合。
3. 每个右括号都有一个对应的相同类型的左括号。

示例 1：
输入：s = "()"
输出：true
示例 2：
输入：s = "()[]{}"
输出：true
示例 3：
输入：s = "(]"
输出：false
function isValid(s) {
    const stack = [];
    const mapping = { ')': '(', '}': '{', ']': '[' };

    for (const char of s) {
        if (mapping[char]) {
            // 右括号：检查栈顶是否匹配
            const top = stack.length ? stack.pop() : '#';
            if (mapping[char] !== top) return false;
        } else {
            // 左括号入栈
            stack.push(char);
        }
    }

    return stack.length === 0;
}

// 测试
console.log(isValid("()"));      // true
console.log(isValid("()[]{}"));  // true
console.log(isValid("(]"));      // false

3. 最小路径和 
求二位矩阵中最小路径和, 从左上角到右下角的路径，使得路径上的数字总和为最小。
限定往右或者往下走。
输入: 
[
[1,3,1],
[1,5,1],
[4,2,1] 
] 
输出: 7
解释: 因为路径 1→3→1→1→1 的总和最小。
function minPathSum(grid) {
    const m = grid.length, n = grid[0].length;
    const dp = Array.from({ length: m }, () => new Array(n).fill(0));

    dp[0][0] = grid[0][0];

    // 初始化第一行
    for (let j = 1; j < n; j++) dp[0][j] = dp[0][j - 1] + grid[0][j];

    // 初始化第一列
    for (let i = 1; i < m; i++) dp[i][0] = dp[i - 1][0] + grid[i][0];

    // 填充其余格子
    for (let i = 1; i < m; i++) {
        for (let j = 1; j < n; j++) {
            dp[i][j] = grid[i][j] + Math.min(dp[i - 1][j], dp[i][j - 1]);
        }
    }

    return dp[m - 1][n - 1];
}

// 测试
console.log(minPathSum([[1,3,1],[1,5,1],[4,2,1]])); // 7
4."4. 两个字符串相加
别名：大数相加
描述：给两个大整数, 用字符串表示, 比如”21543655"", ""4332656442"", 都可能超过1万位, 写一个函数输出他们之和. 需要自己实现加法过程, 不能用某些语言自带的高精度加法函数.
输入：""21543655”, ""4332656442"" 
输出：""4354200097"""
function addStrings(num1, num2) {
    let i = num1.length - 1;
    let j = num2.length - 1;
    let carry = 0;
    let result = '';

    while (i >= 0 || j >= 0 || carry) {
        const a = i >= 0 ? Number(num1[i]) : 0;
        const b = j >= 0 ? Number(num2[j]) : 0;

        const total = a + b + carry;
        carry = Math.floor(total / 10);
        result = String(total % 10) + result;

        i--;
        j--;
    }

    return result;
}

// 测试
console.log(addStrings("21543655", "4332656442")); // "4354200097"
5. 合并区间 
输入： [[1,3],[2,6],[8,10],[15,18]] 
输出：[[1,6],[8,10],[15,18]] 
解释：区间 [1,3] 和 [2,6] 重叠, 将它们合并为 [1,6]."
只出现一次的数字
function merge(intervals) {
    if (!intervals.length) return [];

    intervals.sort((a, b) => a[0] - b[0]);
    const merged = [intervals[0]]; //[[1,3]]

    for (let i = 1; i < intervals.length; i++) {
        const [start, end] = intervals[i];
        if (start <= merged[merged.length - 1][1]) {
            // 有重叠，合并
            merged[merged.length - 1][1] = Math.max(merged[merged.length - 1][1], end);
        } else {
            merged.push([start, end]);
        }
    }

    return merged;
}

// 测试
console.log(merge([[1,3],[2,6],[8,10],[15,18]])); // [[1,6],[8,10],[15,18]]
6.某个元素只出现一次以外，其余每个元素均出现两次，找出1次的
输入：nums = [2,2,1]
输出：1"、.
function singleNumber(nums) {
    return nums.reduce((acc, num) => acc ^ num, 0);
}

// 测试
console.log(singleNumber([2, 2, 1]));        // 1
console.log(singleNumber([4, 1, 2, 1, 2])); // 4
